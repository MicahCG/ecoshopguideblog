## PROJECT OVERVIEW

**Goal:** Build a conversion-optimized storefront that reduces Pinterest → Purchase friction from 3 steps to 2 steps.

**Tech Stack:**
- Platform: Replit (Node.js/Next.js)
- Payment: Stripe Checkout
- Fulfillment: Shopify (backend inventory/orders)
- Domain: shop.ecoshopguide.com
- Design: Match existing Webflow aesthetic

**Key Success Metrics:**
- Single-click purchase experience
- Trust signals visible above fold
- AX-optimized for AI agents
- Mobile-first responsive design
- Page load < 2 seconds

### 4.2 Shopify API Integration Code
```javascript
// /lib/shopify.js
import Shopify from 'shopify-api-node';

const shopify = new Shopify({
  shopName: process.env.SHOPIFY_STORE_DOMAIN,
  accessToken: process.env.SHOPIFY_ADMIN_ACCESS_TOKEN,
});

export async function getProducts(limit = 30) {
  try {
    const products = await shopify.product.list({ limit });
    return products.map(product => ({
      id: product.id,
      title: product.title,
      description: product.body_html,
      price: product.variants[0].price,
      image: product.images[0]?.src,
      handle: product.handle,
      variantId: product.variants[0].id,
      inventory: product.variants[0].inventory_quantity,
      tags: product.tags.split(', ')
    }));
  } catch (error) {
    console.error('Shopify API Error:', error);
    return [];
  }
}

export async function getProductByHandle(handle) {
  const product = await shopify.product.list({ handle });
  return product[0];
}
```

### 4.3 Edge Function for Checkout
```javascript
// /pages/api/checkout.js
import Stripe from 'stripe';
import { getProductByHandle } from '../../lib/shopify';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    const { productHandle, quantity = 1 } = req.body;
    
    // Get product from Shopify
    const product = await getProductByHandle(productHandle);
    
    if (!product) {
      return res.status(404).json({ error: 'Product not found' });
    }

    // Create Stripe Checkout Session
    const session = await stripe.checkout.sessions.create({
      payment_method_types: ['card'],
      line_items: [{
        price_data: {
          currency: 'usd',
          product_data: {
            name: product.title,
            images: [product.images[0]?.src],
            description: product.body_html.substring(0, 300),
          },
          unit_amount: Math.round(parseFloat(product.variants[0].price) * 100),
        },
        quantity,
      }],
      mode: 'payment',
      success_url: `${process.env.NEXT_PUBLIC_SITE_URL}/success?session_id={CHECKOUT_SESSION_ID}`,
      cancel_url: `${process.env.NEXT_PUBLIC_SITE_URL}`,
      metadata: {
        shopify_product_id: product.id.toString(),
        shopify_variant_id: product.variants[0].id.toString(),
      },
      shipping_address_collection: {
        allowed_countries: ['US', 'CA', 'GB', 'AU'],
      },
    });

    res.status(200).json({ sessionId: session.id, url: session.url });
  } catch (error) {
    console.error('Checkout error:', error);
    res.status(500).json({ error: 'Checkout failed' });
  }
}
```

### 4.4 Stripe Webhook Handler
```javascript
// /pages/api/webhooks/stripe.js
import { buffer } from 'micro';
import Stripe from 'stripe';
import Shopify from 'shopify-api-node';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);
const shopify = new Shopify({
  shopName: process.env.SHOPIFY_STORE_DOMAIN,
  accessToken: process.env.SHOPIFY_ADMIN_ACCESS_TOKEN,
});

export const config = {
  api: {
    bodyParser: false,
  },
};

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).end();
  }

  const buf = await buffer(req);
  const sig = req.headers['stripe-signature'];

  let event;

  try {
    event = stripe.webhooks.constructEvent(
      buf,
      sig,
      process.env.STRIPE_WEBHOOK_SECRET
    );
  } catch (err) {
    console.error('Webhook signature verification failed:', err.message);
    return res.status(400).send(`Webhook Error: ${err.message}`);
  }

  // Handle successful payment
  if (event.type === 'checkout.session.completed') {
    const session = event.data.object;

    // Create order in Shopify for fulfillment
    try {
      const order = await shopify.order.create({
        line_items: [{
          variant_id: session.metadata.shopify_variant_id,
          quantity: 1,
        }],
        customer: {
          email: session.customer_details.email,
        },
        shipping_address: {
          first_name: session.shipping_details.name.split(' ')[0],
          last_name: session.shipping_details.name.split(' ')[1] || '',
          address1: session.shipping_details.address.line1,
          address2: session.shipping_details.address.line2,
          city: session.shipping_details.address.city,
          province: session.shipping_details.address.state,
          country: session.shipping_details.address.country,
          zip: session.shipping_details.address.postal_code,
        },
        financial_status: 'paid',
        fulfillment_status: 'unfulfilled',
        note: 'Order from shop.ecoshopguide.com via Stripe',
      });

      console.log('Shopify order created:', order.id);
    } catch (error) {
      console.error('Failed to create Shopify order:', error);
      // Still return 200 to acknowledge receipt of webhook
    }
  }

  res.status(200).json({ received: true });
}
```

### 5.2 Product Card Component
```jsx
// /components/Product/ProductCard.jsx
import { useState } from 'react';
import { loadStripe } from '@stripe/stripe-js';
import toast from 'react-hot-toast';

const stripePromise = loadStripe(process.env.NEXT_PUBLIC_STRIPE_PUBLISHABLE_KEY);

export default function ProductCard({ product }) {
  const [loading, setLoading] = useState(false);

  const handleQuickBuy = async () => {
    setLoading(true);
    toast.loading('Redirecting to checkout...');

    try {
      const response = await fetch('/api/checkout', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ 
          productHandle: product.handle,
          quantity: 1 
        }),
      });

      const { url } = await response.json();
      
      // Redirect to Stripe Checkout
      window.location.href = url;
    } catch (error) {
      toast.dismiss();
      toast.error('Checkout failed. Please try again.');
      setLoading(false);
    }
  };

  return (
    <article 
      className="group bg-white rounded-lg overflow-hidden shadow-sm hover:shadow-xl transition-all duration-300"
      itemScope 
      itemType="https://schema.org/Product"
    >
      {/* Product Image */}
      <div className="relative aspect-square overflow-hidden">
        <img
          src={product.image}
          alt={product.title}
          itemProp="image"
          className="w-full h-full object-cover group-hover:scale-105 transition-transform duration-500"
        />
        
        {/* Quick Buy Overlay */}
        <div className="absolute inset-0 bg-black/40 opacity-0 group-hover:opacity-100 transition-opacity duration-300 flex items-center justify-center">
          <button
            onClick={handleQuickBuy}
            disabled={loading}
            className="bg-white text-primary px-8 py-3 rounded-full font-semibold hover:bg-earth-beige transition-colors disabled:opacity-50"
          >
            {loading ? 'Processing...' : 'Quick Buy'}
          </button>
        </div>

        {/* Eco Badge */}
        {product.tags.includes('eco-certified') && (
          <div className="absolute top-4 right-4 bg-primary text-white px-3 py-1 rounded-full text-sm font-semibold">
            🌱 Eco-Certified
          </div>
        )}
      </div>

      {/* Product Info */}
      <div className="p-4">
        <h3 
          className="font-semibold text-lg mb-2 line-clamp-2 group-hover:text-primary transition-colors"
          itemProp="name"
        >
          {product.title}
        </h3>
        
        <p 
          className="text-neutral-600 text-sm mb-3 line-clamp-2"
          itemProp="description"
          dangerouslySetInnerHTML={{ __html: product.description }}
        />

        <div className="flex items-center justify-between">
          <span 
            className="text-2xl font-bold text-primary"
            itemProp="offers"
            itemScope
            itemType="https://schema.org/Offer"
          >
            <span itemProp="price">${product.price}</span>
            <meta itemProp="priceCurrency" content="USD" />

            onClick={handleQuickBuy}
            disabled={loading}
            className="bg-primary text-white px-6 py-2 rounded-full hover:bg-primary-dark transition-colors disabled:opacity-50"
          >
            Buy Now
          </button>
        </div>
      </div>

      {/* Hidden Schema Data for AI Agents */}
      <meta itemProp="brand" content="Eco Shop Guide" />
      <meta itemProp="category" content={product.tags[0]} />
      <link itemProp="url" href={`https://shop.ecoshopguide.com/products/${product.handle}`} />
    </article>
  );
}
```

## TROUBLESHOOTING

### Common Issues:

**"Products not loading"**
- Check Shopify API credentials
- Verify Storefront API access enabled
- Check network tab for API errors

**"Stripe checkout fails"**
- Verify Stripe keys (test vs live)
- Check webhook signing secret
- Monitor Stripe dashboard for errors

**"Orders not appearing in Shopify"**
- Verify webhook received by checking logs
- Check Shopify API permissions
- Ensure variant IDs match

**"Images loading slowly"**
- Optimize images with Next.js Image component
- Use WebP format
- Implement lazy loading
