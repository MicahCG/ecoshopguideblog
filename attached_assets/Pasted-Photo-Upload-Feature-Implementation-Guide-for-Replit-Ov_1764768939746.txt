Photo Upload Feature Implementation Guide for Replit

  Overview

  This adds photo upload capability to product reviews with a separate 
  upload flow to prevent form crashes. Photos upload independently, then
  URLs are submitted with the review.

  ---

  npm install multer @types/multer

  // Configure multer for file uploads
  const uploadDir = path.resolve(import.meta.dirname, "..",
  "attached_assets");

  // Ensure upload directory exists
  if (!fs.existsSync(uploadDir)) {
    fs.mkdirSync(uploadDir, { recursive: true });
  }

  const multerStorage = multer.diskStorage({
    destination: function (_req, _file, cb) {
      cb(null, uploadDir);
    },
    filename: function (_req, file, cb) {
      const uniqueSuffix = Date.now();
      const ext = path.extname(file.originalname);
      const basename = path.basename(file.originalname,
  ext).replace(/[^a-zA-Z0-9]/g, '_');
      cb(null, `${basename}_${uniqueSuffix}${ext}`);
    }
  });

  const upload = multer({
    storage: multerStorage,
    limits: {
      fileSize: 5 * 1024 * 1024, // 5MB per file
    },
    fileFilter: (_req, file, cb) => {
      const allowedTypes = ['image/jpeg', 'image/jpg', 'image/png',
  'image/webp', 'image/heic'];
      if (allowedTypes.includes(file.mimetype)) {
        cb(null, true);
      } else {
        cb(new Error('Only image files (JPEG, PNG, WebP, HEIC) are 
  allowed'));
      }
    }
  });

  Add upload endpoint (right after the /api/products GET endpoint):

  // Upload photos for reviews
  app.post("/api/upload/review-photos", upload.array('photos', 5), async

  0) {

      }

      // Generate URLs for uploaded files
      const photoUrls = req.files.map(file =>

      res.status(500).json({

        message: error.message
      });
    }
  });

  Step 4: Update ReviewForm Component 
  (client/src/components/ReviewForm.tsx)

  Replace the entire file with this:

  import { useState, useRef } from 'react';
  import { useForm } from 'react-hook-form';
  import { zodResolver } from '@hookform/resolvers/zod';
  import { z } from 'zod';
  import { Star, Loader2, X, Upload } from 'lucide-react';
  import { Button } from '@/components/ui/button';
  import { Card, CardContent } from '@/components/ui/card';
  import {
    Form,
    FormControl,
    FormField,
    FormItem,
    FormLabel,
    FormMessage,
  } from '@/components/ui/form';
  import { Input } from '@/components/ui/input';
  import { Textarea } from '@/components/ui/textarea';
  import { useToast } from '@/hooks/use-toast';
  import { apiRequest, queryClient } from '@/lib/queryClient';

  const reviewSchema = z.object({
    rating: z.number().int().min(1, 'Please select a rating').max(5),
    title: z.string().min(5, 'Title must be at least 5 characters'),
    comment: z.string().min(20, 'Review must be at least 20 characters'),
    userName: z.string().min(2, 'Name must be at least 2 characters'),
    userEmail: z.string().email('Please enter a valid 
  email').optional().or(z.literal('')),
    photos: z.array(z.string()).optional(),
  });

  type ReviewFormData = z.infer<typeof reviewSchema>;

  interface ReviewFormProps {
    productId: string;
    onSuccess?: () => void;
    initialRating?: number;
  }

  export default function ReviewForm({ productId, onSuccess, initialRating 
  }: ReviewFormProps) {
    const [hoveredRating, setHoveredRating] = useState(0);
    const [uploadedPhotos, setUploadedPhotos] = useState<string[]>([]);
    const [isUploading, setIsUploading] = useState(false);
    const fileInputRef = useRef<HTMLInputElement>(null);
    const { toast } = useToast();

    const form = useForm<ReviewFormData>({
      resolver: zodResolver(reviewSchema),
      defaultValues: {
        rating: initialRating || (undefined as any),
        title: '',
        comment: '',
        userName: '',
        userEmail: '',
        photos: [],
      },
    });

    const selectedRating = form.watch('rating');

    const handleFileSelect = async (event:
  React.ChangeEvent<HTMLInputElement>) => {
      const files = event.target.files;
      if (!files || files.length === 0) return;

      // Check if adding these files would exceed the limit
      if (uploadedPhotos.length + files.length > 5) {
        toast({
          title: 'Too Many Photos',
          description: 'You can upload a maximum of 5 photos',
          variant: 'destructive',
        });
        return;
      }

      setIsUploading(true);

      try {
        const formData = new FormData();
        Array.from(files).forEach(file => {
          formData.append('photos', file);
        });

        const response = await fetch('/api/upload/review-photos', {
          method: 'POST',
          body: formData,
        });

        if (!response.ok) {
          throw new Error('Upload failed');
        }

        const data = await response.json();
        const newPhotos = [...uploadedPhotos, ...data.urls];
        setUploadedPhotos(newPhotos);
        form.setValue('photos', newPhotos);

        toast({
          title: 'Photos Uploaded',
          description: `${files.length} photo(s) uploaded successfully`,
        });
      } catch (error: any) {
        toast({
          title: 'Upload Failed',
          description: error.message || 'Failed to upload photos',
          variant: 'destructive',
        });
      } finally {
        setIsUploading(false);
        if (fileInputRef.current) {
          fileInputRef.current.value = '';
        }
      }
    };

    const removePhoto = (index: number) => {
      const newPhotos = uploadedPhotos.filter((_, i) => i !== index);
      setUploadedPhotos(newPhotos);
      form.setValue('photos', newPhotos);
    };

    const onSubmit = async (data: ReviewFormData) => {
      try {
        const response = await apiRequest('POST',
  `/api/products/${productId}/reviews`, data);

        if (!response.ok) {
          const error = await response.json();
          throw new Error(error.message || 'Failed to submit review');
        }

        // Invalidate reviews query to refetch
        await queryClient.invalidateQueries({
          queryKey: ['/api/products', productId, 'reviews']
        });

        toast({
          title: 'Review Submitted',
          description: 'Thank you for your feedback!',
        });

        form.reset();
        setUploadedPhotos([]);
        onSuccess?.();
      } catch (error: any) {
        toast({
          title: 'Submission Failed',
          description: error.message,
          variant: 'destructive',
        });
      }
    };

    return (
      <Card>
        <CardContent className="p-4 sm:p-6">
          <Form {...form}>
            <form onSubmit={form.handleSubmit(onSubmit)} 
  className="space-y-4 sm:space-y-6">
              {/* Rating */}
              <FormField
                control={form.control}
                name="rating"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Rating *</FormLabel>
                    <FormControl>
                      <div className="flex gap-1">
                        {[1, 2, 3, 4, 5].map((rating) => (
                          <button
                            key={rating}
                            type="button"
                            onClick={() => field.onChange(rating)}
                            onMouseEnter={() => setHoveredRating(rating)}
                            onMouseLeave={() => setHoveredRating(0)}
                            className="transition-opacity hover:opacity-80"
                            data-testid={`button-rating-${rating}`}
                          >
                            <Star
                              className={`w-8 h-8 sm:w-10 sm:h-10 ${
                                rating <= (hoveredRating || selectedRating)
                                  ? 'fill-primary text-primary'
                                  : 'text-muted'
                              }`}
                            />
                          </button>
                        ))}
                      </div>
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}
              />

              {/* Name and Email */}
              <div className="grid sm:grid-cols-2 gap-4">
                <FormField
                  control={form.control}
                  name="userName"
                  render={({ field }) => (
                    <FormItem>
                      <FormLabel>Your Name *</FormLabel>
                      <FormControl>
                        <Input
                          placeholder="John Doe"
                          {...field}
                          data-testid="input-reviewer-name"
                        />
                      </FormControl>
                      <FormMessage />
                    </FormItem>
                  )}
                />

                <FormField
                  control={form.control}
                  name="userEmail"
                  render={({ field }) => (
                    <FormItem>
                      <FormLabel>Your Email <span 
  className="text-muted-foreground 
  font-normal">(optional)</span></FormLabel>
                      <FormControl>
                        <Input
                          type="email"
                          placeholder="john@example.com"
                          {...field}
                          data-testid="input-reviewer-email"
                        />
                      </FormControl>
                      <FormMessage />
                    </FormItem>
                  )}
                />
              </div>

              {/* Title */}
              <FormField
                control={form.control}
                name="title"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Review Title *</FormLabel>
                    <FormControl>
                      <Input
                        placeholder="Summarize your experience"
                        {...field}
                        data-testid="input-review-title"
                      />
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}
              />

              {/* Comment */}
              <FormField
                control={form.control}
                name="comment"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>Your Review *</FormLabel>
                    <FormControl>
                      <Textarea
                        placeholder="Share your experience with this 
  product..."
                        rows={5}
                        {...field}
                        data-testid="textarea-review-comment"
                      />
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}
              />

              {/* Photo Upload */}
              <div className="space-y-3">
                <FormLabel>
                  Add Photos <span className="text-muted-foreground 
  font-normal">(optional, max 5)</span>
                </FormLabel>

                {uploadedPhotos.length > 0 && (
                  <div className="grid grid-cols-2 sm:grid-cols-3 gap-3">
                    {uploadedPhotos.map((photo, index) => (
                      <div key={index} className="relative group">
                        <img
                          src={photo}
                          alt={`Review photo ${index + 1}`}
                          className="w-full h-24 object-cover rounded-lg 
  border"
                        />
                        <button
                          type="button"
                          onClick={() => removePhoto(index)}
                          className="absolute top-1 right-1 bg-destructive
  text-destructive-foreground rounded-full p-1 opacity-0
  group-hover:opacity-100 transition-opacity"
                          aria-label="Remove photo"
                        >
                          <X className="w-4 h-4" />
                        </button>
                      </div>
                    ))}
                  </div>
                )}

                {uploadedPhotos.length < 5 && (
                  <div>
                    <input
                      ref={fileInputRef}
                      type="file"
                      
  accept="image/jpeg,image/jpg,image/png,image/webp,image/heic"
                      multiple
                      onChange={handleFileSelect}
                      className="hidden"
                      disabled={isUploading}
                    />
                    <Button
                      type="button"
                      variant="outline"
                      onClick={() => fileInputRef.current?.click()}
                      disabled={isUploading}
                      className="w-full sm:w-auto"
                    >
                      {isUploading ? (
                        <>
                          <Loader2 className="mr-2 h-4 w-4 animate-spin" />
                          Uploading...
                        </>
                      ) : (
                        <>
                          <Upload className="mr-2 h-4 w-4" />
                          Upload Photos
                        </>
                      )}
                    </Button>
                    <p className="text-xs text-muted-foreground mt-2">
                      Supported formats: JPEG, PNG, WebP, HEIC (max 5MB
  each)
                    </p>
                  </div>
                )}
              </div>

              <Button
                type="submit"
                disabled={form.formState.isSubmitting}
                className="w-full sm:w-auto min-h-11 py-2.5"
                data-testid="button-submit-review"
              >
                {form.formState.isSubmitting ? (
                  <>
                    <Loader2 className="mr-2 h-4 w-4 animate-spin" />
                    Submitting...
                  </>
                ) : (
                  'Submit Review'
                )}
              </Button>
            </form>
          </Form>
        </CardContent>
      </Card>
    );
  }
